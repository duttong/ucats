%{
	#include <stdio.h>
	#include <stdlib.h>
	#include <unistd.h>
	#include "nortlib.h"
	#include "globmsg.h"
	#include "addresses.h"
    #include "valves.h"
%}

State initialize  {
	     > Telemetry Start
	+15  > Mode GC Init
	 +5  > ssv goto position 1
	     > sample cal
}

%{
#define N2fail  40		// psi		(below this value set ECDs cold)
%}

Partition

%{
// the array "Sequence" can be any length. Sequence contains the
// sequence of SSV positions for a calibration run.  Running caldoit
// and starting the cal mode will call these routines.

unsigned int Sequence[] = {1,1,3,3,5,5,7,7,9,9};
unsigned int NUMCALSEQ = sizeof(Sequence)/sizeof(unsigned int);
unsigned int SeqIndex = 0;
%}

State CalSSV {
	depending on ( Injecting1 ) {
		Validate NextCal;
	}
}

State NextCal {
	+72 { switch_ssv( CALSSVADD, Sequence[SeqIndex] );
		 if ( ++SeqIndex >= NUMCALSEQ ) SeqIndex = 0;
	   }
	+2 Validate CalSSV;
}

#  This partition monitors the N2 carrier gas.  If the N2 regulator
#  pressure on the lowside is below N2fail the ECDs will be set
#  to a cold setting.
%{
unsigned int N2lowCount = 0;
unsigned int MAXN2LOWCNTS = 10;
%}
Partition
State CheckCarrier {
	+10	{ if ( (convert(presL_N2) < N2fail) && (convert(presL_N2) > 0) ) {
			msg( -2, "** (%d) LOW N2 Pressure (presL_N2 = %f)", N2lowCount, convert(presL_N2) );
			if ( ++N2lowCount >= MAXN2LOWCNTS ) 
				Validate SetECDsCold; 
		  }
		  else if ( convert(presL_N2) >= N2fail ) 
			N2lowCount = 0; 
		}
}

State SetECDsCold {
	+1	> Omega change SP1 CH1 ECD 10
	+1  > Omega change SP1 CH2 ECD 10
}


#  This partition looks at the digin bit InitShutdown (see
#  digital.tmc).  Once set, the Shutdown proceedure is called.
Partition
State ShutdownComputer {
	+1	{ if ( InitShutdown == FALSE ) validate Shutdown; }
}

State Shutdown {
	> Shutdown
}
